//membuat database baru
CREATE DATABASE pegawaisib5;

//MENGGUNAKAN DATABASE
use pegawaisib5;

//membuat table 
CREATE TABLE jenis(
    id int PRIMARY KEY AUTO_INCREMENT,
    nama varchar(30) NOT NULL UNIQUE
    );

/*
id adalah nama kolom atau header kolom
int adalah type database
PRIMARY KEY adalah key untuk aoutmatisasi penomoran
NOT NULL menandakan kolom tersebut tidak boleh kosong jika data mula diisi
UNIQUE adalah key untuk kolom agar tidak boleh sama */


//membuat tabel relasi
CREATE TABLE produk (
    id int PRIMARY KEY AUTO_INCREMENT,
    nama varchar(30) NOT NULL UNIQUE,
    harga double NOT NULL,
    stok int NOT NULL,
    idjenis int NOT NULL REFERENCES jenis(id),
    foto varchar(30));
